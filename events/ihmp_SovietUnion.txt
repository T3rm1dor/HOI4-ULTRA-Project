country_event = {
	id = amm_soviet.1
	title = amm_soviet.1.t
	desc = amm_soviet.1.d
	picture = GFX_report_event_dead_soldiers
	fire_only_once = yes
	is_triggered_only = yes
	
	immediate = {

			add_ideas = sov_chock

		}

		effect_tooltip = {
			add_ideas = sov_chock
			add_ideas = sov_chock_air
		}
	}
}
