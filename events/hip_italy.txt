			}

				is_core_of = ITA
				is_on_continent = europe
				is_controlled_by = CAN
				NOT = {state = 114}
			}			
			any_owned_state = {
				is_core_of = ITA
				is_on_continent = europe				
				is_controlled_by = USA
				NOT = {state = 114}
			}
			any_owned_state = {
				is_core_of = ITA
				is_on_continent = europe				
				is_controlled_by = FRA
				NOT = {state = 114}
			}			
		}
	}
		
	mean_time_to_happen = {
		days = 14
	}	
	
	immediate = {
		kill_country_leader = yes
	}
	
	option = {
		name = hip_italy.1.a
		ai_chance = { factor = 100 }
		create_country_leader = {
			name = "Pietro Badoglio"
			desc = "POLITICS_PIETRO_BADOGLIO_DESC"
			picture = "Portrait_Italy_Pietro_Badoglio.dds"
			expire = "1965.1.1"
			ideology = conservatism
			traits = {
			}
		}
		set_politics = {
			ruling_party = democratic 
			elections_allowed = yes
		}
		add_popularity = {
			ideology = democratic 
			popularity = 0.35
		}
		add_popularity = {  
			ideology = communism
			popularity = 0.15
		}
		add_popularity = {
			ideology = fascism
			popularity = -0.10
		}			
		if = {
			limit = {
				ENG = { exists = yes }
				ENG = { is_faction_leader = yes }
			}			
			ENG = { country_event = { id = hip_italy.2 days = 21 } }
		}
		if = {
			limit = {
				USA = { exists = yes }
				USA = { is_faction_leader = yes }
			}			
			USA = { country_event = { id = hip_italy.2 days = 21 } }
		}		
	}
}
